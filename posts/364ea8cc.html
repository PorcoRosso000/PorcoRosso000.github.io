1.饿汉式
/**
 * 饿汉式
 * 类加载到内存后，就实例化一个单例，JVM保证线程安全
 * 简单实用，推荐使用！
 * 唯一缺点：不管用到与否，类装载时就完成实例化
 * Class.forName("")
 * （话说你不用的，你装载它干啥）
 */
public class Mgr01 &#123;
    private static final Mgr01 INSTANCE = new Mgr01();
    
    //上面也可以写成
     /*private static final Mgr02 INSTANCE;
       static &#123;
        INSTANCE = new Mgr02();
       &#125;
    */
    private Mgr01() &#123;&#125;;

    public static Mgr01 getInstance() &#123;
        return INSTANCE;
    &#125;

    public static void main(String[] args) &#123;
        Mgr01 m1 = Mgr01.getInstance();
        Mgr01 m2 = Mgr01.getInstance();
        System.out.println(m1 == m2);

2.懒汉式
/**
 * lazy loading
 * 也称懒汉式
 * 虽然达到了按需初始化的目的，但却带来线程不安全的问题
 */
public class Mgr03 &#123;
    private static Mgr03 INSTANCE;

    private Mgr03() &#123;
    &#125;
    //有线程安全问题的原因是 两个线程a,b a县城通过判空 if (INSTANCE == null) 进入大括号内的逻辑中但是没有 执行到 INSTANCE = new Mgr03(); 这时候b通过判断条件也进来了,然后b执行了new 方法  a又执行了一次new就导致线程不安全,就创建了两个不同的对象 就不是单例了
    public static Mgr03 getInstance() &#123;
        if (INSTANCE == null) &#123;
            try &#123;
                Thread.sleep(1);
            &#125; catch (InterruptedException e) &#123;
                e.printStackTrace();
            &#125;
            INSTANCE = new Mgr03();
        &#125;
        return INSTANCE;
    &#125;


    public static void main(String[] args) &#123;
        //创建了100个线程去执行
        for(int i=0; i<100; i++) &#123;

//上面的 Mgr08 的定义利用的enum是一种特殊的class.代码中的第一行INSTANCE会被编译器编译为 Mgr08 本身的一个对象.当第一次访问 Mgr08.INSTANCE时会创建该对象,并且enum变量的创建是线程安全的

    &#125;
    public int getValue() &#123;

    &#125;
    public void setValue(int value) &#123;
        this.value = value;
    &#125;
&#125;
//调用

SingletonEnum singleton = SingletonEnum.INSTANCE;
System.out.println(singleton.getValue());
