模糊分页条件查询

@PostMapping("/list")
@SystemLog("mongo查询系统用户列表")
public Result<PageResult<SysUser>> list(@RequestBody SysUserRequest sysUserRequest) &#123;
    log.info("功能名称：mongo查询系统用户列表，请求路径：【&#123;&#125;】，请求方式：【&#123;&#125;】，请求参数：【&#123;&#125;】", request.getRequestURI(),
            request.getMethod(), JSONObject.toJSONString(sysUserRequest));
    // 封装查询条件
    Query query = new Query();
    if (StringUtils.isNotEmpty(sysUserRequest.getSex())) &#123;
        query.addCriteria(Criteria.where("sex").is(sysUserRequest.getSex()));
    &#125;
    // 根据用户名模糊查询
    if (StringUtils.isNotEmpty(sysUserRequest.getUsername())) &#123;
        Pattern pattern = Pattern.compile("^.*" + sysUserRequest.getUsername() + ".*$");
        query.addCriteria(Criteria.where("username").regex(pattern));
    &#125;
    // 区间
    if (sysUserRequest.getStartTime() != null && sysUserRequest.getEndTime() != null) &#123;
        query.addCriteria(Criteria.where("createTime").gte(sysUserRequest.getStartTime()).lte(sysUserRequest.getEndTime()));
    &#125; else if (sysUserRequest.getStartTime() != null) &#123;
        query.addCriteria(Criteria.where("createTime").gte(sysUserRequest.getStartTime()));
    &#125; else if (sysUserRequest.getEndTime() != null) &#123;
        query.addCriteria(Criteria.where("createTime").lte(sysUserRequest.getEndTime()));
    &#125;
    // 计算总记录数
    long total = mongoTemplate.count(query, SysUser.class);
    // 分页
    // 跳过几条数据
    query.skip((sysUserRequest.getPageNum() - 1) * sysUserRequest.getPageSize());
    // 设置每页记录数
    query.limit(sysUserRequest.getPageSize());
    // 根据创建时间降序
    query.with(Sort.by(Sort.Order.desc("createTime")));
    // 查询操作
    List<SysUser> list = mongoTemplate.find(query, SysUser.class);
    Result<PageResult<SysUser>> result = PageResult.toResult(total, list);
    log.info("功能名称：mongo查询系统用户列表，请求路径：【&#123;&#125;】，请求方式：【&#123;&#125;】，响应结果：【&#123;&#125;】", request.getRequestURI(),
            request.getMethod(), JSONObject.toJSONString(result));
    return result;

public class AuthFilter implements GlobalFilter, Ordered &#123;

    /**
     * 过滤逻辑
     * @param exchange  请求上下文对象
     * @param chain  过滤器链
     * @return
     */
    @Override
    public Mono<Void> filter(ServerWebExchange exchange, GatewayFilterChain chain) &#123;
        // 1：通过请求上下文对象 获取请求对象
        ServerHttpRequest request = exchange.getRequest();
        // 2: 获取请求的URI
        String path = request.getURI().getPath();
        // 3: 获取请求的方式
        HttpMethod method = exchange.getRequest().getMethod();
        // 4: 通过log4j记录请求日志
        log.info("请求的路径是：[&#123;&#125;]，请求的方式是：[&#123;&#125;]", path, method);
        return chain.filter(exchange);  // 放行

mongo微服务添加日志记录

@PostMapping("/list")
public Result list(@RequestBody SysUserRequest sysUserRequest) &#123;
    String requestURI = request.getRequestURI();
    String method = request.getMethod();
    String reqJson = JSON.toJSONString(sysUserRequest);
    log.info("本次请求的路径是：[&#123;&#125;]，请求方式是：[&#123;&#125;]，请求参数是：[&#123;&#125;]", requestURI, method, reqJson);
    // 构建查询条件
    Criteria criteria = new Criteria();
    if (StringUtils.isNotEmpty(sysUserRequest.getSex())) &#123;
       criteria.and("sex").is(sysUserRequest.getSex());
    &#125;
    if (StringUtils.isNotEmpty(sysUserRequest.getUsername())) &#123;
        Pattern pattern = Pattern.compile("^.*"+ sysUserRequest.getUsername() +".*$");
        criteria.and("username").regex(pattern);
    &#125;
    if (sysUserRequest.getStart() != null) &#123;
        criteria.and("age").gt(sysUserRequest.getStart());
    &#125;
    if (sysUserRequest.getEnd() != null) &#123;
        criteria.and("age").lt(sysUserRequest.getEnd());
    &#125;
    // 添加查询条件
    Query query = new Query();
    query.addCriteria(criteria);
    // 查询总记录数
    long count = mongoTemplate.count(query, SysUser.class);
    // 分页查询
    // 指定从MongoDB中读取的记录条数
    query.limit(sysUserRequest.getPageSize());
    // 指定跳过指定数量的数据
    query.skip((sysUserRequest.getPageNum() - 1) * sysUserRequest.getPageSize());
    List<SysUser> users = mongoTemplate.find(query, SysUser.class);
    String respJson = JSON.toJSONString(users);
    log.info("本次请求的路径是：[&#123;&#125;]，请求方式是：[&#123;&#125;]，响应数据是：[&#123;&#125;]", requestURI, method, respJson);
    return PageResult.toResult(count, users);

编写AOP全局记录日志

@Component
@Aspect
@Log4j2
public class LogAop &#123;
    @Autowired
    private SysLogService sysLogService;
    @Around("@annotation(sysLog)")
    public Object doLog(ProceedingJoinPoint pjp, SystemLog sysLog) &#123;
        SysLog slog = new SysLog();
        // 1: 获取请求对象
        ServletRequestAttributes servletRequestAttributes = (ServletRequestAttributes) RequestContextHolder.getRequestAttributes();
        HttpServletRequest request = servletRequestAttributes.getRequest();
        // 2: 获取请求方式 请求路径
        String requestURI = request.getRequestURI();
        String method = request.getMethod();
        slog.setRequestURI(requestURI);
        slog.setRequestMethod(method);
        // 3：统计方法执行时间
        long s = System.currentTimeMillis();
        Object res = null;
        try &#123;
            res = pjp.proceed();
        &#125; catch (Throwable throwable) &#123;
            throwable.printStackTrace();
        &#125;
        long e = System.currentTimeMillis();
        long requestTime = e- s;
        slog.setRequestTime(requestTime);
        // 4：获取请求的类名方法名
        String name = pjp.getTarget().getClass().getName();
        String methodName = pjp.getSignature().getName();
        slog.setClazzName(name);
        slog.setMethodName(methodName);
        // 5: 获取功能名称
        String value = sysLog.value(); // 功能名称
        slog.setFunctionDesc(value);
        // 6： 请求响应 参数
        Object[] args = pjp.getArgs();
        String reqJson = "";
        String respJson = "";
        if (args.length > 0) &#123;
            // 请求参数
            reqJson = JSON.toJSONString(args[0]);
        &#125;
        if (res != null) &#123;
            respJson = JSON.toJSONString(res);
        &#125;
        slog.setReqJson(reqJson);
        slog.setRespJson(respJson);
        sysLogService.addLog(slog);
        log.info("本次请求的功能是：[&#123;&#125;]，请求路径：[&#123;&#125;]，请求方式：[&#123;&#125;]，类名：[&#123;&#125;]，方法名：[&#123;&#125;]，请求参数：[&#123;&#125;]，响应参数：[&#123;&#125;]，请求时间：[&#123;&#125;]",
                value, requestURI, method, name, methodName, reqJson, respJson, requestTime);
        return res;
    &#125;
