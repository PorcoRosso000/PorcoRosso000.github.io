### 23种设计模式

#### 单例模式

##### 1.饿汉式

```java
/**
 * 饿汉式
 * 类加载到内存后，就实例化一个单例，JVM保证线程安全
 * 简单实用，推荐使用！
 * 唯一缺点：不管用到与否，类装载时就完成实例化
 * Class.forName("")
 * （话说你不用的，你装载它干啥）
 */
public class Mgr01 {
    private static final Mgr01 INSTANCE = new Mgr01();
    
    //上面也可以写成
     /*private static final Mgr02 INSTANCE;
       static {
        INSTANCE = new Mgr02();
       }
    */
    private Mgr01() {};

    public static Mgr01 getInstance() {
        return INSTANCE;
    }

    public static void main(String[] args) {
        Mgr01 m1 = Mgr01.getInstance();
        Mgr01 m2 = Mgr01.getInstance();
        System.out.println(m1 == m2);
    }
}
```

##### 2.懒汉式

```java
/**
 * lazy loading
 * 也称懒汉式
 * 虽然达到了按需初始化的目的，但却带来线程不安全的问题
 */
public class Mgr03 {
    private static Mgr03 INSTANCE;

    private Mgr03() {
    }
    //有线程安全问题的原因是 两个线程a,b a县城通过判空 if (INSTANCE == null) 进入大括号内的逻辑中但是没有 执行到 INSTANCE = new Mgr03(); 这时候b通过判断条件也进来了,然后b执行了new 方法  a又执行了一次new就导致线程不安全,就创建了两个不同的对象 就不是单例了
    public static Mgr03 getInstance() {
        if (INSTANCE == null) {
            try {
                Thread.sleep(1);
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
            INSTANCE = new Mgr03();
        }
        return INSTANCE;
    }


    public static void main(String[] args) {
        //创建了100个线程去执行
        for(int i=0; i<100; i++) {
            new Thread(()->
                System.out.println(Mgr03.getInstance().hashCode())
            ).start();
        }
    }
}

//为了解决线程安全的问题加了锁,但是加锁之后效率不高
public class Mgr04 {

//为了解决效率问题,减小加锁的范围,减少同步代码块,发现有现成安全问题
public class Mgr05 {
    private static Mgr05 INSTANCE;

    private Mgr05() {
    }

    public static Mgr05 getInstance() {
        if (INSTANCE == null) {
            //尝试通过减小同步代码块的方式提高效率，然后不可行有线程安全问题原因和上述线程安全原因一致
            synchronized (Mgr05.class) {
                try {
                    Thread.sleep(1);
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
                INSTANCE = new Mgr05();
            }
        }
        return INSTANCE;
    }


    public static void main(String[] args) {
        for(int i=0; i<100; i++) {
            new Thread(()->{
                System.out.println(Mgr05.getInstance().hashCode());
            }).start();
        }
    }
}


//为了解决减小同步代码块,提升锁效率之后的线程安全问题再增加一层判断,双重判断,这样饿汉式没有问题的单例就出来了
public class Mgr06 {
    /*
       * 利用静态变量来记录 Mgr06 的唯一实例
       * volatile 关键字确保：当 INSTANCE 变量被初始化成 Mgr06 实例时，
       * 多个线程正确地处理 INSTANCE 变量
       *
       */
    private static volatile Mgr06 INSTANCE; 
    /*
       * 构造器私有化，只有 Mgr06 类内才可以调用构造器
       */
    private Mgr06() {
    }
	/*
       *
       * 检查实例，如果不存在，就进入同步区域
       */
    public static Mgr06 getInstance() {
        if (INSTANCE == null) {
            //双重检查
            synchronized (Mgr06.class) {
                //进入同步区域之后再检查一次
                if(INSTANCE == null) {
                    try {
                        Thread.sleep(1);
                    } catch (InterruptedException e) {
                        e.printStackTrace();
                    }
                    INSTANCE = new Mgr06();
                }
            }
        }
        return INSTANCE;
    }


    public static void main(String[] args) {
        for(int i=0; i<100; i++) {
            new Thread(()->{
                System.out.println(Mgr06.getInstance().hashCode());
            }).start();
        }
    }
}



/**
 * 静态内部类方式
 * JVM保证单例
 * 加载外部类时不会加载内部类，这样可以实现懒加载
 */
public class Mgr07 {

    private Mgr07() {
    }

    private static class Mgr07Holder {
        private final static Mgr07 INSTANCE = new Mgr07();
    }

    public static Mgr07 getInstance() {
        return Mgr07Holder.INSTANCE;
    }

    public static void main(String[] args) {
        for(int i=0; i<100; i++) {
            new Thread(()->{
                System.out.println(Mgr07.getInstance().hashCode());
            }).start();
        }
    }

}
```

##### 3.枚举单例

```java
/**
  * 不仅可以解决线程同步，还可以防止反序列化。
  * 使用枚举的方式实现单例
   */
  public enum Mgr08 {

  INSTANCE;

  public void doSomething() {
      System.out.println("doSomething");
  }

  public static void main(String[] args) {
      for(int i=0; i<100; i++) {
          new Thread(()->{
              System.out.println(Mgr08.INSTANCE.hashCode());
              //调用方式
              Mgr08.INSTANCE.doSomething()；
          }).start();
      }
  }

}
//上面的 Mgr08 的定义利用的enum是一种特殊的class.代码中的第一行INSTANCE会被编译器编译为 Mgr08 本身的一个对象.当第一次访问 Mgr08.INSTANCE时会创建该对象,并且enum变量的创建是线程安全的


//例：
public enum SingletonEnum {
    INSTANCE;
    int value;
    // 这里我们可以自定义构造函数.
    private SingletonEnum() {
        value = 1;
        System.out.println("INSTANCE now created!");
    }
    public int getValue() {
        return value;
    }
    public void setValue(int value) {
        this.value = value;
    }
}
//调用
System.out.println("start get instance!");
SingletonEnum singleton = SingletonEnum.INSTANCE;
System.out.println(singleton.getValue());
singleton.setValue(2);
System.out.println(singleton.getValue());
//结果
start get instance!
INSTANCE now created!
1
2
```
